}, [theme]);

const [particles, setParticles] = useState([]);

const triggerShine = () => {
  const burst = Array.from({ length: 20 }).map((_, i) => ({
    id: i,
    x: Math.random() * 240 - 120,
    y: Math.random() * -240,
    size: Math.random() * 6 + 6,
  }));

  setParticles(burst);
  setTimeout(() => setParticles([]), 1000);
};


<button
  onClick={() => {
    triggerShine();
    setShowModal(false);
  }}
>
  Enter Gainly
</button>


<div className="particle-layer">
  {particles.map((p) => (
    <span
      key={p.id}
      className="particle"
      style={{
        transform: `translate(${p.x}px, ${p.y}px)`,
        width: p.size,
        height: p.size,
      }}
    />
  ))}
</div>


.particle-layer {
  position: absolute;
  inset: 0;
  pointer-events: none;
}

.particle {
  position: absolute;
  top: 50%;
  left: 50%;
  background: var(--accent);
  border-radius: 50%;
  animation: pop 1s ease-out forwards;
}

@keyframes pop {
  from {
    transform: translate(0, 0) scale(1);
    opacity: 1;
  }
  to {
    transform: translate(var(--x), var(--y)) scale(0);
    opacity: 0;
  }
}
